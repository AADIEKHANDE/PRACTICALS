            # List of students who play both cricket and badminton
            def cri_bad():
                uni_cri_bad=[]

            # list of students who play either cricket or badminton but not both
            def cri_bad_not_both():
                not_cri_bad = []

                        not_cri_bad.append(i)
                for j in groupB:
                    if j not in groupA:
                        not_cri_bad.append(j)
